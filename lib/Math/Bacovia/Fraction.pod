=encoding utf8

=head1 NAME

Math::Bacovia::Fraction - Represents a symbolic fraction.

=head1 SYNOPSIS

    use Math::Bacovia qw(Fraction Symbol);

    my $f = Fraction(Symbol('a'), Symbol('b'));
    my $g = Fraction(Symbol('x'), Symbol('y'));

    say (($f + $g)->pretty);    #=> (((a * y) + (x * b))/(b * y))
